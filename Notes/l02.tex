\textbf{Basic geometric definitions (for LP in higher dimension):}
Suppose we live in $\mathbb{R}^d$ for some $d \in \mathbb{N}$.
\begin{itemize}
\item \emph{hyperplane} is $d-1$ dimensional subspace  $\{\mathbf{x} \in \mathbb{R}^d: \textbf{a} \cdot \mathbf{x} = c    \}$,  where $a \in \mathbb{R}^d$ and $c \in \mathbb{R}$.
\item \emph{closed halfspace} is  $\{\mathbf{x} \in \mathbb{R}^d: \textbf{a} \cdot \mathbf{x} \leq c    \}$,  where $a \in \mathbb{R}^d$ and $c \in \mathbb{R}$.
\item $C \subseteq \mathbb{R}^d$ is \emph{convex} if $\forall \mathbf{x},\textbf{y} \in C, \forall t \in [0,1], t\mathbf{x}+(1-t)\textbf{y} \in C$. (line between $x$ and $y$ is in $C$)  \\

\end{itemize}

\emph{Convex hull} of a set $X = conv(X)$ is the intersection of all convex sets containing $X$.

Examples: bunch of points, point + line

\textbf{Claim:}
$\mathbf{x} \in conv(X)$ iff $\mathbf{x}$ is a convex combination of points from $X$.\\
That is: 
\[
\exists n, \exists \mathbf{x}_1,\ldots,\mathbf{x}_n \in X, \exists t_1,\ldots,t_n \in [0,\infty), \sum t_i = 1, x = \sum t_i\mathbf{x}_i 
\]
\textbf{Proof:} (as exercise)
Hints: consider 
\[
Y = \left\{\sum t_i\mathbf{x}_i : n\in \mathbb{N}, \mathbf{x}_1,\ldots,\mathbf{x}_n \in X,  t_1,\ldots,t_n \in [0,\infty), \sum t_i = 1    \right\}
\]
and show that $Y \subseteq conv(X)$, that $X \subseteq Y$ and  $Y$ is convex.
\vskip 15em
\solution{
$Y \subseteq conv(X)$:  clear for $n=1,2$, use projection and induction on $n$: 
\begin{align*}
\textbf{y} &= \sum_{i=1}^{n}t_i \mathbf{x}_i \\
               &= t_1\mathbf{x}_1 + (1-t_1) \sum_{i=2}^{n}\frac{t_i}{1-t_1} \mathbf{x}_i
\end{align*}
Now use that second part of the sum $\sum_{i=2}^{n}\frac{t_i}{1-t_1} \mathbf{x}_i \in conv(X)$ by induction.

$X \subseteq Y$ is clear from definition


$Y$ is convex: verify definition of convexity
\begin{align*}
\mathbf{x} &= \sum t_i \mathbf{x}_i     &  \textbf{x}' &= \sum t'_i \mathbf{x}_i 
\end{align*}
Let $t \in [0,1]$:
\begin{align*}
t \mathbf{x} + (1-t) \mathbf{x}' = \sum (t t_i + (1-t)t'_i) \mathbf{x}_i 
\end{align*}
Check that $ \sum (t t_i + (1-t)t') =  t\sum t  +  (1-t)  \sum t'_i = 1$.
}
